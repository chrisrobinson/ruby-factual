  def test_filtering
    row = @table.filter(:two_letter_abbrev => 'WA').find_one
    assert_equal row["state"].value, "Washington"

    row = @table.filter(:two_letter_abbrev => { '$has' => 'a' }).sort(:state => 1).find_one
    assert_equal row["state"].value, "California"
  end
