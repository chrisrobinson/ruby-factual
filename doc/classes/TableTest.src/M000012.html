  def test_sorting
    row = @table.sort(:state => 1).find_one
    assert_equal row["state"].value, "California"

    assert_raise Factual::ApiError do 
      # secondary sort will be supported in next release
      row = @table.sort({:state => 1}, {:test_field1 => 1}).find_one
      row["state"].value
    end
  end
