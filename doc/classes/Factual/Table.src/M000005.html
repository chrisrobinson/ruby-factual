    def each_row
      resp = @adapter.read_table(@table_key, @filters, @sorts, @page_size, @page)

      @total_rows = resp["total_rows"]
      rows = resp["data"]

      rows.each do |row_data|
        subject_key = row_data.shift
        row = Row.new(self, subject_key, row_data) 
        yield(row) if block_given?
      end
    end
